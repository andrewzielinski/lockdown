                        def user_groups_assignable_for_user(usr)
                                return [] if usr.nil?

                                if administrator?(usr)
                                        UserGroup.find(:all, :order => :name)
                                else
                                        UserGroup.find_by_sql "select user_groups.* from user_groups, user_groups_users\nwhere user_groups.id = user_groups_users.user_group_id\nand user_groups_users.user_id = \#{usr.id}\norder by user_groups.name\n"
                                end
                        end
